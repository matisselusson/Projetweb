
DROP TABLE IF EXISTS temps_passes;
DROP TABLE IF EXISTS tickets;
DROP TABLE IF EXISTS projets_collaborateurs;
DROP TABLE IF EXISTS contrats;
DROP TABLE IF EXISTS projets;
DROP TABLE IF EXISTS clients;
DROP TABLE IF EXISTS utilisateurs;



CREATE TABLE utilisateurs (
    id           INT AUTO_INCREMENT PRIMARY KEY,
    nom          VARCHAR(100) NOT NULL,
    prenom       VARCHAR(100) NOT NULL,
    email        VARCHAR(255) NOT NULL UNIQUE,
    mot_de_passe VARCHAR(255) NOT NULL,
    role         VARCHAR(50)  NOT NULL DEFAULT 'collaborateur',
    cree_le      DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE clients (
    id        INT AUTO_INCREMENT PRIMARY KEY,
    nom       VARCHAR(255) NOT NULL,
    email     VARCHAR(255),
    telephone VARCHAR(20),
    cree_le   DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE projets (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    titre       VARCHAR(255) NOT NULL,
    description TEXT,
    priorite    VARCHAR(50)  NOT NULL DEFAULT 'Moyenne',
    statut      VARCHAR(50)  NOT NULL DEFAULT 'Nouveau',
    client_id   INT NOT NULL,
    cree_le     DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (client_id) REFERENCES clients(id)
);

CREATE TABLE projets_collaborateurs (
    projet_id      INT NOT NULL,
    utilisateur_id INT NOT NULL,
    PRIMARY KEY (projet_id, utilisateur_id),
    FOREIGN KEY (projet_id)      REFERENCES projets(id),
    FOREIGN KEY (utilisateur_id) REFERENCES utilisateurs(id)
);

CREATE TABLE contrats (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    projet_id       INT NOT NULL UNIQUE,
    heures_incluses FLOAT NOT NULL DEFAULT 0,
    taux_horaire    FLOAT DEFAULT NULL,
    date_debut      DATE,
    date_fin        DATE,
    cree_le         DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (projet_id) REFERENCES projets(id)
);

CREATE TABLE tickets (
    id           INT AUTO_INCREMENT PRIMARY KEY,
    titre        VARCHAR(255) NOT NULL,
    description  TEXT,
    priorite     VARCHAR(50)  NOT NULL DEFAULT 'Moyenne',
    statut       VARCHAR(50)  NOT NULL DEFAULT 'Nouveau',
    type         VARCHAR(50)  NOT NULL DEFAULT 'Inclus',
    projet_id    INT NOT NULL,
    assigne_id   INT,
    temps_estime FLOAT DEFAULT NULL,
    cree_le      DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (projet_id)  REFERENCES projets(id),
    FOREIGN KEY (assigne_id) REFERENCES utilisateurs(id)
);

CREATE TABLE temps_passes (
    id             INT AUTO_INCREMENT PRIMARY KEY,
    ticket_id      INT NOT NULL,
    utilisateur_id INT NOT NULL,
    duree          FLOAT NOT NULL,
    date_saisie    DATE NOT NULL,
    commentaire    TEXT,
    cree_le        DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (ticket_id)      REFERENCES tickets(id),
    FOREIGN KEY (utilisateur_id) REFERENCES utilisateurs(id)
);
